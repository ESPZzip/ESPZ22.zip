function removeItem() { const query = document.getElementById("removeItemId").value.trim().toLowerCase(); if (!query) { alert("Debes escribir el ID o nombre del ítem."); return; } let items = getItems(); const index = items.findIndex(it => it.id.toLowerCase() === query || it.name.toLowerCase() === query); if (index === -1) { alert("No se encontró ningún ítem con ese ID o nombre."); return; } const removed = items.splice(index, 1)[0]; saveItems(items); renderMarketplace(); logAction(`Se eliminó el ítem '${removed.name}' (${removed.id}) del Marketplace`); alert(`Ítem '${removed.name}' eliminado correctamente.`); }
